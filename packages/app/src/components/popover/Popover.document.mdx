# Popover 

Popover / Layer 를 표현할 수 있는 컴포넌트 입니다.

# Usages

```tsx
import { QueuePopover } from '@legacy/components/popover/Popover';

export const SomeComponent = () => {
  const [open, setOpen] = useState(false);

  return (
    <QueuePopover.Root
      open={open}
      onOpenChange={(open) => {
        setOpen(open);
      }}
    >
      <QueuePopover.Anchor>
        <QueueButton
          size={QUEUE_UI_SIZE.MEDIUM}
          color={QUEUE_UI_COLOR.RED}
          onClick={() => setOpen(true)}
        >
          open
        </QueueButton>
      </QueuePopover.Anchor>
      <QueuePopover.Content>
        <QueueButton 
          size={QUEUE_UI_SIZE.MEDIUM} 
          color={QUEUE_UI_COLOR.RED}
          onClick={() => setOpen(false)}
        >
          close
        </QueueButton>
      </QueuePopover.Content>
    </QueuePopover.Root>
  )
}
```
