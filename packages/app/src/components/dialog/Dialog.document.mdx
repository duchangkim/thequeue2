# Dialog

다이얼로그를 표현할 수 있는 컴포넌트입니다.

# Usages

먼저 다이얼로그로 사용할 컴포넌트를 선언합니다.

```tsx
import { QueueButton } from 'components/buttons/button/Button';
import { QueueDialog, QueueDialogRootRef } from 'components/dialog/Dialog';
import { QueueInput } from 'components/input/Input';
import { useRef, useState } from 'react';
import { useTranslation } from 'react-i18next';
import { QUEUE_UI_COLOR } from 'styles/ui/Color';
import { QUEUE_UI_SIZE } from 'styles/ui/Size';

export const SomeDialog = () => {
  const dialogRef = useRef<QueueDialogRootRef>();

  return (
    <QueueDialog.Root ref={dialogRef}>
      <QueueDialog.Title>
        다이얼로그의 타이틀 
      </QueueDialog.Title>
      <QueueDialog.Description>
        다이얼로그의 메인 컨텐츠
      </QueueDialog.Description>
      <QueueDialog.Footer>
        <QueueButton
          size={QUEUE_UI_SIZE.MEDIUM}
          color={QUEUE_UI_COLOR.RED}
          onClick={() => dialogRef.current.close()}
        >
          닫기
        </QueueButton>
        <QueueButton 
          size={QUEUE_UI_SIZE.MEDIUM}
          color={QUEUE_UI_COLOR.BLUE}
        >
          확인
        </QueueButton>
      </QueueDialog.Footer>
    </QueueDialog.Root>
  )
};
```

그리고 RootRenderer 를 사용하여 이를 호출합니다.

```tsx
import { useRootRenderer } from 'cdk/root-renderer/root-renderer';
import { SomeDialog } from './SomeDialog';

export const SomeComponent = () => {
  const rootRenderer = useRootRenderer();

  const openDialog = () => {
    rootRenderer.render(
      <SomeDialog />
    );
  }

  return (
    <div>
      <button onClick={openDialog}>
      다이얼로그 열기
      </button>
    </div>
  )
}
```
