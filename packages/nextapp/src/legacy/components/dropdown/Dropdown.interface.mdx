# QueueDropdown API

## QueueDropdown Component Interface

```tsx
// Queue에서 간략하게 만들 구조
<QueueDropdown {...rootProps}>
  <QueueDropdown.Trigger />

  <QueueDropdown.Content {...contentProps}>
    <QueueDropdown.Item />

    <QueueDropdown.Group label="" {...groupProps}>
      <QueueDropdown.Item />
    </QueueDropdown.Group>

    <QueueDropdown.CheckboxItem {...checkboxProps} />

    <QueueDropdown.RadioGroup label="" {...radioGroupProps}>
      <QueueDropdown.RadioItem {...radioProps} />
    </QueueDropdown.RadioGroup>

    <QueueDropdown.Sub {...subProps}>
      <QueueDropdown.SubTrigger />

      <QueueDropdown.SubContent {...subContentProps} />
    </QueueDropdown.Sub>
    
    <QueueDropdown.Separator />
  </QueueDropdown.Content>
</QueueDropdown>
```
